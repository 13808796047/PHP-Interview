/**
 * 从数组中找出不重复的数字
 * @param $array
 */
function find($array)
{
    if(empty($array) || !is_array($array)) {
        return [];
    }

    $data = [];

    foreach ($array as $key=>$value) {

        if(isset($data[$value])) {
            $data[$value]++;
        }else{
            $data[$value] = 1;
        }
    }
    $result = [];
    foreach ($data as $k=>$v) {
        if($v != 1) {
            unset($data[$k]);
        }else{
            $result[] = $k;
        }

    }
    return $result;
}

function bullsort($array) 
{
  if(empty($array) || !is_array($array)){
    return false;
  }
  $length = count($array);
  for($i = 1; $i < $length;$i++)
  {
    for($j = 0;$j<$length-$i;$j++)
    {
      if($array[$j+1] < $array[$j]) 
      {
        $temp = $array[$j];
        $array[$j] = $array[$j+1];
        $array[$j+1] = $temp;
      }
    }
  }
  return $array;
}
